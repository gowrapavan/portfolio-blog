-- 1. Enable the pgvector extension to work with embeddings
create extension if not exists vector;

-- 2. Create the table to store your Dev Logs
create table dev_logs (
  id uuid primary key default gen_random_uuid(),
  content text not null,               -- The AI-generated log summary
  git_diff text,                       -- The raw code changes
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  embedding vector(1536)               -- The mathematical "meaning" of the log
);

-- 3. Create a function so the Chatbot can search these logs later
create or replace function match_dev_logs (
  query_embedding vector(1536),
  match_threshold float,
  match_count int
)
returns table (
  id uuid,
  content text,
  similarity float
)
language plpgsql
as $$
begin
  return query
  select
    dev_logs.id,
    dev_logs.content,
    1 - (dev_logs.embedding <=> query_embedding) as similarity
  from dev_logs
  where 1 - (dev_logs.embedding <=> query_embedding) > match_threshold
  order by dev_logs.embedding <=> query_embedding
  limit match_count;
end;
$$;
